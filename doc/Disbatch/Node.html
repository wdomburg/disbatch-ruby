def register
  return true if @open

  doc = Mongo.try do
    Disbatch.db[:nodes].find_and_modify({
      :query  => { :_id => @id, :pid => nil },
      :update => { :$set => {
        :pid          => Process.pid,
        :version      => Disbatch::VERSION,
        :spec_version => Disbatch::SPEC_VERSION
      } }
    })
  end

  if doc.nil?
    false
  else
    @open = true
  end
end

def release
  return false unless @open

  doc = Mongo.try do
    Disbatch.db[:nodes].find_and_modify({
      :query  => { :_id => @id, :pid => Process.pid },
      :update => { :$set => {
        :pid   => nil,
        :mtime => Time.now,
        :atime => Time.now
      } }
    })
  end

  if doc.nil? 
    false
  else
    @open = false
    true
  end
end

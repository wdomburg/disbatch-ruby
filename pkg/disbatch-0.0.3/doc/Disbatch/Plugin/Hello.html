def execute(task)
  return nil unless defined?(task) && task.respond_to?(:params)

  name = task.params['name']
  len  = rand(5)

  task.log("Hello, #{name}!")
  sleep(len)
  task.log("Goodbye, #{name}. It was a pleasure spending #{len} seconds with you.")

  task.conclude
end
